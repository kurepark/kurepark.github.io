function storyGradientHeightCheck(element){
        var winWidth = $(window).width();
        var maxWebTextHeight = 640; //웹 글길이 기준
        var viewWebTextHeight = 480; //기준 초과시 보여줄 그라데이션 높이값
 
        var maxMobTextHeight = 560; //모바일 글길이 기준
        var viewMobTextHeight = 420; //기준 초과시 보여줄 그라데이션 높이값
 
        element.each(function(){
 
             var targetNode = $(this).find('.froala-body');
             var storyBody = $(this).children('.story-body');
             var boxHeight = storyBody.height();
 
             if(winWidth < 669) { //모바일
                 if(boxHeight > maxMobTextHeight){
                    storyBody.css({
                        'height':(viewMobTextHeight + 10) +'px',
                        'overflow' : 'hidden',
                        'transition':'height .3s'
                    });
                    makeGradientDom(targetNode);
                 }
             } else{
                if(boxHeight > maxWebTextHeight){
                    storyBody.css({
                        'height':(viewWebTextHeight + 10) +'px',
                        'overflow' : 'hidden',
                        'transition':'height .3s'
                    });
                    makeGradientDom(targetNode);
                }
             }
        });
    }
    function makeGradientDom(node){
        node.after("<div class='summary-wrapper'><div class='summary-gradient'></div><button type='button' class='more-btn' onclick='closeStoryGradient($(this))'>더보기</button></div>");
    }
    function closeStoryGradient(node){
        var gradientWrapper = node.parent();
        var storyBody = gradientWrapper.parent().parent('.story-body');
        var textBody = gradientWrapper.siblings('.froala-body');
        var textBodyHeight = textBody.outerHeight();
 
        storyBody.css('height',textBodyHeight +20);
        node.css('display','none');
        node.siblings('.summary-gradient').css({
            'opacity':0,
            'visiblity' : 'hidden'
        });
        setTimeout(function(){
            gradientWrapper.remove();
        },700);
    }
